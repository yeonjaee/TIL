```python
n = int(input())
cyc = 0				# 사이클 수
num = n

while True:
    a = int(num/10) #새로운 수의 앞자리
    b = num % 10
    s = (a+b) % 10 #새로운 수의 뒷자리
    num = b*10 + s
    cyc = cyc + 1

    
    if (num == n):
        break
print(cyc)
```
